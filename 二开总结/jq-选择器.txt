语法 	                描述   
$(this) 	        当前 HTML 元素
$("p") 	                所有 <p> 元素
$("p.intro") 	        所有 class="intro" 的 <p> 元素
$(".intro") 	        所有 class="intro" 的元素
$("#intro") 	        id="intro" 的元素
$("ul li:first") 	每个 <ul> 的第一个 <li> 元素
$("[href$='.jpg']") 	所有带有以 ".jpg" 结尾的属性值的 href 属性
$("div#intro .head") 	id="intro" 的 <div> 元素中的所有 class="head" 的元素
